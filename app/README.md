

# You should have 
	node, bower, gulp installed

# Install node packages using
 	npm install

# Install the client side packages using
	bower install
    in case if that does not work use
    sudo bower install --allow-root



# Start application with normal files, it will use files  from public/ folder on localhost:3000
 	node ./bin/www 
 	or
 	npm start

# OR

# In order to watch file changes and reload server use the command
# It creates a build/ folder and will server the files from buil/ folder in minified and uglifed version on localhost:5000
	gulp serve
